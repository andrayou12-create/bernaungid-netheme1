<script>
    function kirimWA() {
        const nama = document.getElementById('namaTamu').value;
        if (nama === "") {
            alert("Silakan masukkan nama tamu terlebih dahulu!");
            return;
        }

        // 1. MENGAMBIL URL DASAR SECARA OTOMATIS
        // Kode ini akan otomatis mengambil domain dan folder (latif-erna)
        const currentUrl = window.location.origin + window.location.pathname;
        const linkFix = currentUrl.replace('tamu.html', 'index.html') + "?to=" + encodeURIComponent(nama);

        // 2. SUSUN TEKS PESAN
        const pesanMentah = 
            "Kepada Yth.\n" +
            "Bapak/Ibu/Saudara/i\n" +
            "*" + nama + "*\n" + 
            "_______\n\n" +
            "Assalamualaikum Warahmatullahi Wabarakatuh\n\n" +
            "Tanpa mengurangi rasa hormat, perkenankan kami mengundang Bapak/Ibu/Saudara/i, teman sekaligus sahabat, untuk menghadiri acara pernikahan kami.\n\n" +
            linkFix + "\n\n" +
            "NB: Hasil maksimal buka lewat browser chrome/safari\n\n" +
            "Merupakan suatu kebahagiaan bagi kami apabila Bapak/Ibu/Saudara/i berkenan untuk hadir dan memberikan doa restu.\n\n" +
            "Wassalamualaikum Warahmatullahi Wabarakatuh\n\n" +
            "Terima Kasih\n\n" +
            "Hormat kami,\n" +
            "Latif & Erna\n" + 
            "________\n\n" +
            "Mohon maaf jika ada kesalahan penulisan nama atau gelar";

        // 3. ENCODE & BUKA WHATSAPP
        const pesanFull = encodeURIComponent(pesanMentah);
        window.open("https://api.whatsapp.com/send?text=" + pesanFull, '_blank');
    }
</script>
